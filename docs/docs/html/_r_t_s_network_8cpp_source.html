21inline void insertRegionInMapAndToProcess(const networkOfTA::NetworkRegion &reg,
22 std::deque<NetworkRegionPtr> &toProcess,
23 std::unordered_set<networkOfTA::NetworkRegion, networkOfTA::NetworkRegionHash> &regionsHashMap,
24 const std::vector<std::unordered_map<std::string, int>> &clocksIndices,
25 const std::vector<absl::flat_hash_map<int, std::vector<timed_automaton::ast::clockConstraint>>> &invariants,
26 const bool isInvariantFree)
27{
28 bool isRegionLegal = true;
29
30 if (!isInvariantFree)
31 {
32 const auto &regRegs = reg.getRegions();
33
34 for (int i = 0; i < static_cast<int>(regRegs.size()); i++)
35 if (invariants[i].contains(regRegs[i].getLocation()))
36 if (!isInvariantSatisfied(invariants[i].at(regRegs[i].getLocation()), regRegs[i].getClockValuation(), clocksIndices[i]))
37 {
38 isRegionLegal = false;
39 break;
40 }
41 }
42
43 if (isRegionLegal)
44 {
45 // ReSharper disable once CppTooWideScopeInitStatement
46 auto [iter, inserted] = regionsHashMap.insert(reg);
47
48 // Only add to toProcess if it's a new region.
49 if (inserted)
50 toProcess.push_back(&*iter);
51 }
52}

70inline bool checkIfTargetRegionReached(const std::vector<region::Region> &currentRegionRegions,
71 const std::vector<std::optional<int>> &currentTargetLocations,
72 const std::vector<std::vector<timed_automaton::ast::clockConstraint>> &currentGoalClockConstraints,
73 const absl::btree_map<std::string, int> &currentNetworkVariables,
74 const std::vector<timed_automaton::ast::clockConstraint> &intVarConstr,
75 const std::vector<std::unordered_map<std::string, int>> &clocksIndices)
76{
77 // With symmetry reduction, targetLocations has been permuted with regions during canonicalization,
78 // so we can directly check each region against its corresponding target location.
79 for (int i = 0; i < static_cast<int>(currentTargetLocations.size()); i++)
80 {
81 if (currentTargetLocations[i].has_value())
82 {
83 if (currentRegionRegions[i].getLocation() != currentTargetLocations[i])
84 return false;
85 }
86 }
87
88 // Checking whether constraints on integer variables are satisfied.
89 if (!intVarConstr.empty())
90 {
91 for (const auto &integerValue: intVarConstr)
92 {
93 if (!integerValue.isSatisfied(currentNetworkVariables.at(integerValue.clock), false))
94 return false;
95 }
96 }
97
98 // Checking whether constraints on clocks are satisfied.
99 // With symmetry reduction, goalClockConstraints has been permuted with regions during canonicalization,
100 // so we can directly check each region against its corresponding clock constraints.
101 if (!currentGoalClockConstraints.empty())
102 {
103 for (int i = 0; i < static_cast<int>(currentGoalClockConstraints.size()); i++)
104 {
105 if (!currentGoalClockConstraints[i].empty())
106 {
107 const auto &cv = currentRegionRegions[i].getClockValuation();
108
109 for (const auto &clockConstraint: currentGoalClockConstraints[i])
110 {
111 // ReSharper disable once CppTooWideScopeInitStatement
112 const int clockIdx = clocksIndices[i].at(clockConstraint.clock);
113
114 if (!clockConstraint.isSatisfied(cv[clockIdx].first, cv[clockIdx].second))
115 return false;
116 }
117 }
118 }
119 }
120
121 return true;
122}

126std::vector<networkOfTA::NetworkRegion> networkOfTA::RTSNetwork::forwardReachability(
127 const std::vector<timed_automaton::ast::clockConstraint> &intVarConstr,
128 const std::vector<std::vector<timed_automaton::ast::clockConstraint>> &goalClockConstraints,
129 const std::vector<std::optional<int>> &targetLocs,
130 const ssee explorationTechnique) const
131{
132 // Starting the timer for measuring computation.
133 const auto start = std::chrono::high_resolution_clock::now();
134
135 // Initializing auxiliary data structures for reachability computation.
136 std::deque<NetworkRegionPtr> toProcess{};
137 std::unordered_set<NetworkRegion, NetworkRegionHash> regionsHashMap{};
138
139 const bool useSymmetryReduction = !symmetryGroups.empty() && symmetryReduction;
140
141 // Apply symmetry reduction to initial regions if enabled
142 for (const auto &init: getInitialRegions())
143 {
144 NetworkRegion regionToInsert = init.clone();
145
146 // Setting target locations for network regions.
147 regionToInsert.setTargetLocations(targetLocs);
148
149 // Setting goal clock constraints for network regions.
150 regionToInsert.setGoalClockConstraints(goalClockConstraints);
151
152 // Apply canonical form if symmetry reduction is enabled.
153 if (useSymmetryReduction)
154 regionToInsert = regionToInsert.getCanonicalForm(symmetryGroups);
155
156 auto [iter, inserted] = regionsHashMap.insert(regionToInsert);
157 toProcess.push_back(&*iter);
158 }
159
160
161 unsigned long long int totalRegions = 0;
162
163 while (!toProcess.empty())
164 {
165 const NetworkRegion &currentRegion = explorationTechnique == BFS ? *toProcess.front() : *toProcess.back();
166
167#ifdef RTSNETWORK_DEBUG
168
169 std::cout << "Current region:\n" << currentRegion.toString() << std::endl;
170
171 std::cout << "Locations of the network region:" << std::endl;
172 for (const auto &reg: currentRegion.getRegions())
173 std::cout << reg.getLocation() << " ";
174 std::cout << "\n\n";
175
176#endif
177
178 // Getting the regions of the network region currentRegion.
179 const auto &currentRegionRegions = currentRegion.getRegions();
180
181 const bool isTargetRegionReached = checkIfTargetRegionReached(currentRegionRegions,
182 currentRegion.getTargetLocations(),
183 currentRegion.getGoalClockConstraints(),
184 currentRegion.getNetworkVariables(),
185 intVarConstr,
186 clocksIndices);
187
188 if (isTargetRegionReached)
189 {
190 // Ending the timer for measuring computation.
191 const auto end = std::chrono::high_resolution_clock::now();
192
193 std::cout << "Goal is reachable\n";
194 std::cout << "Number of regions: " << totalRegions << std::endl;
195
196 const auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start);
197 std::cout << "Total time : " << duration.count() << " microseconds." << std::endl;
198
199#ifdef EARLY_EXIT
200
201 std::quick_exit(EXIT_SUCCESS);
202
203#endif
204
205 return { currentRegion };
206 }
207
208 // Computing a network immediate delay successor if no region is in an urgent location.
209 const bool isDelayComputable = !std::ranges::any_of(automataWithUrgentLocations, [&](const auto &pair) {
210 return pair.second.contains(currentRegionRegions[pair.first].getLocation());
211 });
212
213 const NetworkRegion delaySuccessor = isDelayComputable ? currentRegion.getImmediateDelaySuccessor(maxConstants) : NetworkRegion{};
214
215 // Setting up the transitions for the network discrete successor computation.
216 std::vector<std::reference_wrapper<const std::vector<transition>>> transitions{};
217 transitions.reserve(currentRegionRegions.size());
218 for (int i = 0; i < static_cast<int>(currentRegionRegions.size()); i++)
219 transitions.emplace_back(std::cref(outTransitions[i][currentRegionRegions[i].getLocation()]));
220
221 // Computing network discrete successors.
222 const std::vector<NetworkRegion> discreteSuccessors = currentRegion.getImmediateDiscreteSuccessors(transitions, clocksIndices, locationsToInt);
223
224 totalRegions += discreteSuccessors.size() + (isDelayComputable ? 1 : 0);
225
226 // Removing now since we do not need it anymore.
227 explorationTechnique == BFS ? toProcess.pop_front() : toProcess.pop_back();
228
229 // We insert the delay successor first and then the discrete successors.
230 // Apply canonical form if symmetry reduction is enabled
231 if (isDelayComputable)
232 {
233 if (useSymmetryReduction)
234 {
235 // ReSharper disable once CppTooWideScopeInitStatement
236 const NetworkRegion canonicalDelaySucc = delaySuccessor.getCanonicalForm(symmetryGroups);
237 insertRegionInMapAndToProcess(canonicalDelaySucc, toProcess, regionsHashMap, clocksIndices, invariants, isInvariantFree);
238 } else
239 insertRegionInMapAndToProcess(delaySuccessor, toProcess, regionsHashMap, clocksIndices, invariants, isInvariantFree);
240 }
241
242 for (const auto &discreteSuccessor: discreteSuccessors)
243 {
244 if (useSymmetryReduction)
245 {
246 // ReSharper disable once CppTooWideScopeInitStatement
247 const NetworkRegion canonicalDiscSucc = discreteSuccessor.getCanonicalForm(symmetryGroups);
248 insertRegionInMapAndToProcess(canonicalDiscSucc, toProcess, regionsHashMap, clocksIndices, invariants, isInvariantFree);
249 } else
250 insertRegionInMapAndToProcess(discreteSuccessor, toProcess, regionsHashMap, clocksIndices, invariants, isInvariantFree);
251 }
252 }
253
254 // No target region has been reached if the while loop ends.
255 // Ending the timer for measuring computation.
256 const auto end = std::chrono::high_resolution_clock::now();
257
258 std::cout << "Goal is not reachable\n";
259 std::cout << "Number of regions: " << totalRegions << std::endl;
260
261 const auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start);
262 std::cout << "Total time : " << duration.count() << " microseconds." << std::endl;
263
264#ifdef EARLY_EXIT
265
266 std::quick_exit(EXIT_SUCCESS);
267
268#endif
269
270 return {};
271}
