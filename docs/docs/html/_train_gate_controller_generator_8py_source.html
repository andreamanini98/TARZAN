71def escape_uppaal(s: Optional[str]) -> Optional[str]:
72 if s is None:
73 return None
74 s2 = s.replace("<=", "&lt;=").replace(">=", "&gt;=").replace("<", "&lt;").replace(">", "&gt;")

87def evaluate_cnt_conjunct(s: str, cntv: Optional[int]) -> Optional[bool]:
88 """
89 Evaluate a simple cnt conjunct like 'cnt == 1' or 'cnt > 1' using numeric cntv.
90 Returns True/False if can evaluate; None if unrecognized or cntv is None.
91 """
92 if cntv is None:
93 return None
94 q = s.replace(" ", "")
95 # equality
96 m = re.match(r'cnt==(\d+)', q)
97 if m:
98 return cntv == int(m.group(1))
99 m = re.match(r'cnt>(\d+)', q)
100 if m:
101 return cntv > int(m.group(1))
102 m = re.match(r'cnt<(\d+)', q)
103 if m:
104 return cntv < int(m.group(1))
105 m = re.match(r'cnt>=(\d+)', q)
106 if m:
107 return cntv >= int(m.group(1))
108 m = re.match(r'cnt<=(\d+)', q)
109 if m:
110 return cntv <= int(m.group(1))
111 m = re.match(r'cnt!=(\d+)', q)
112 if m:
113 return cntv != int(m.group(1))
114 return None

116def sanitize_guard_for_modeB(guard: Optional[str], cntv: Optional[int], train_index: Optional[int]) -> Tuple[bool, Optional[str]]:
117 """
118 For modeB: evaluate cnt conjuncts (if any) against cntv.
119 - If any cnt-conjunct evaluates to False => transition not allowed (return (False, None))
120 - Otherwise remove cnt-conjuncts from guard and return the escaped guard without them.
121 For modeA: simply return (True, escaped guard)
122 train_index: if provided, will be used to replace 'x' with 'x{train_index}' in the remaining guard
123 """
124 if guard is None or guard.strip() == "":
125 return True, None
126 parts = split_conjuncts(guard)
127 remaining = []
128 for p in parts:
129 if is_cnt_conjunct(p):
130 val = evaluate_cnt_conjunct(p, cntv)
131 if val is False:
132 return False, None
133 # if val is True or None (cannot evaluate but contains cnt), we treat:
134 # - if val is True: drop the cnt conjunct
135 # - if val is None: **conservative**: treat as not allowed (since cnt unknown) -> but we want to allow if modeA only
136 # Here in flattening we will assume cntv is always provided when modeB; if it's None we disallow
137 if val is None:
138 return False, None
139 # else val==True -> skip adding to remaining (we remove cnt conjunct)
140 continue
141 else:
142 remaining.append(p)
143 # now remaining contains non-cnt conjuncts; replace x if needed and escape
144 replaced = []
145 for r in remaining:
146 rr = r
147 if train_index is not None:
148 rr = re.sub(r"\bx\b", f"x{train_index}", rr)
149 rr = escape_uppaal(rr)
150 if rr:
151 replaced.append(rr)
152 if not replaced:
153 return True, None
154 return True, " and ".join(replaced)

164def normalize_assign(assign: Optional[str], train_idx: Optional[int], modeB: bool) -> Optional[str]:
165 """
166 Normalize assignment text:
167 - remove cnt assignments entirely when modeB==True
168 - replace x -> x{train_idx} when train_idx provided
169 - split separators ';' or ',' and rejoin with ', '
170 """
171 if not assign:
172 return None
173 parts = [p.strip() for p in assign.replace(";", ",").split(",") if p.strip()]
174 out = []
175 for p in parts:
176 # strip cnt updates in modeB
177 if modeB and re.search(r'\bcnt\b', p):
178 continue
179 if train_idx is not None:
180 p = re.sub(r"\bx\b", f"x{train_idx}", p)
181 out.append(p)
182 # deduplicate preserving order
183 seen = set()
184 dedup = []
185 for a in out:
186 key = a.replace(" ", "")
187 if key not in seen:
188 seen.add(key)
189 dedup.append(a)
190 return ", ".join(dedup) if dedup else None

192def join_assigns(*assigns: Optional[str]) -> Optional[str]:
193 parts = []
194 for a in assigns:
195 if a:
196 parts.append(a)
197 if not parts:
198 return None
199 out = []
200 seen = set()
201 for p in parts:
202 for q in [x.strip() for x in p.split(",") if x.strip()]:
203 key = q.replace(" ", "")
204 if key not in seen:
205 seen.add(key)
206 out.append(q)
207 return ", ".join(out) if out else None

210 if not sync:

212 m = re.match(r'\s*([A-Za-z_]\w*)\s*([!?])\s*', sync)

217def interpret_cnt_assignment(assign: Optional[str]) -> Optional[Tuple[str, int]]:
218 """
219 Interpret cnt-related assignment in controller transitions.
220 Returns:
221 - ("abs", k) if 'cnt = k'
222 - ("delta", d) if 'cnt = cnt + 1' or 'cnt = cnt - 1'
223 - None if no cnt assignment recognized.
224 """
225 if not assign:
226 return None
227 s = assign.replace(" ", "")
228 m_abs = re.search(r'cnt=(\d+)', s)
229 if m_abs:
230 return ("abs", int(m_abs.group(1)))
231 if 'cnt=cnt+1' in s:
232 return ("delta", 1)
233 if 'cnt=cnt-1' in s:
234 return ("delta", -1)
235 if 'cnt=0' in s:
236 return ("abs", 0)
237 return None

242def build_states(N: int, modeB: bool):
243 gate_names = [s for s, _ in GATE_STATES]
244 ctrl_names = [s for s, _ in CTRL_STATES]
245 train_names = [s for s, _ in TRAIN_STATES]
246 product_trains = list(product(train_names, repeat=N))
247 states = []
248 for g in gate_names:
249 for c in ctrl_names:
250 for ttuple in product_trains:
251 if modeB:
252 for cnt in range(0, N + 1):
253 states.append((g, c, ttuple, cnt))
254 else:
255 states.append((g, c, ttuple, None))
256 index = {states[i]: i for i in range(len(states))}
257 return states, index

262def flatten_model(N: int, modeB: bool):
263 states, index = build_states(N, modeB)
264 flat_trans = []
265
266 gate_by_src: Dict[str, List[Tuple]] = {}
267 for tr in GATE_TRANS:
268 gate_by_src.setdefault(tr[0], []).append(tr)
269 ctrl_by_src: Dict[str, List[Tuple]] = {}
270 for tr in CTRL_TRANS:
271 ctrl_by_src.setdefault(tr[0], []).append(tr)
272 train_by_src: Dict[str, List[Tuple]] = {}
273 for tr in TRAIN_TRANS:
274 train_by_src.setdefault(tr[0], []).append(tr)
275
276 for s_idx, state in enumerate(states):
277 gstate, cstate, ttuple, cntv = state
278
279 # helper to create non-sync transitions
280 def add_non_sync(src_val, dict_, train_index: Optional[int] = None):
281 for (src, tgt, guard, sync, assign, resetx) in dict_.get(src_val, []):
282 if sync:
283 continue
284 # For modeB: evaluate cnt-conjuncts and remove them from label
285 if modeB:
286 allowed, guard_label = sanitize_guard_for_modeB(guard, cntv, train_index)
287 if not allowed:
288 continue
289 else:
290 guard_label = replace_train_x(guard, train_index)
291 # Assignments: in modeB remove cnt updates using normalize_assign
292 assign_label = normalize_assign(assign, train_index, modeB)
293 # reset train clock if requested
294 if resetx and train_index is not None:
295 assign_label = join_assigns(assign_label, f"x{train_index} = 0")
296 # compute destination cnt when controller transition
297 dst_cnt = cntv
298 if dict_ is ctrl_by_src:
299 if modeB and assign:
300 cnt_effect = interpret_cnt_assignment(assign)
301 if cnt_effect:
302 typ, val = cnt_effect
303 if typ == "abs":
304 dst_cnt = val
305 else:
306 dst_cnt = (cntv if cntv is not None else 0) + val
307 if dst_cnt < 0:
308 dst_cnt = 0
309 if dst_cnt > N:
310 dst_cnt = N
311 # build destination
312 if dict_ is gate_by_src:
313 dst = (tgt, cstate, ttuple, dst_cnt if modeB else None)
314 elif dict_ is ctrl_by_src:
315 dst = (gstate, tgt, ttuple, dst_cnt if modeB else None)
316 else:
317 new_ttuple = list(ttuple)
318 new_ttuple[train_index - 1] = tgt
319 dst = (gstate, cstate, tuple(new_ttuple), dst_cnt if modeB else None)
320 dst_idx = index[dst]
321 flat_trans.append({"from": s_idx, "to": dst_idx, "guard": guard_label, "assign": assign_label})
322
323 # add non-syncs
324 add_non_sync(gstate, gate_by_src, None)
325 add_non_sync(cstate, ctrl_by_src, None)
326 for t_i in range(1, N + 1):
327 add_non_sync(ttuple[t_i - 1], train_by_src, t_i)
328
329 # collect sync transitions (gate, ctrl, trains)
330 sync_pool = []
331 for tr in gate_by_src.get(gstate, []):
332 if tr[3]:
333 sync_pool.append(("gate", None, tr))
334 for tr in ctrl_by_src.get(cstate, []):
335 if tr[3]:
336 sync_pool.append(("ctrl", None, tr))
337 for tid in range(N):
338 local = ttuple[tid]
339 for tr in train_by_src.get(local, []):
340 if tr[3]:
341 sync_pool.append(("train", tid, tr))
342
343 # pairwise synchronization
344 for i in range(len(sync_pool)):
345 for j in range(i + 1, len(sync_pool)):
346 comp1, idx1, tr1 = sync_pool[i]
347 comp2, idx2, tr2 = sync_pool[j]
348 (s1, t1, g1, syn1, a1, r1) = tr1
349 (s2, t2, g2, syn2, a2, r2) = tr2
350 p1 = parse_sync(syn1)
351 p2 = parse_sync(syn2)
352 if p1 is None or p2 is None:
353 continue
354 chan1, dir1 = p1
355 chan2, dir2 = p2
356 if chan1 != chan2:
357 continue
358 if dir1 == dir2:
359 continue
360 # determine new component states
361 new_g, new_c, new_ttuple = gstate, cstate, list(ttuple)
362 if comp1 == "gate":
363 new_g = t1
364 elif comp1 == "ctrl":
365 new_c = t1
366 else:
367 new_ttuple[idx1] = t1
368 if comp2 == "gate":
369 new_g = t2
370 elif comp2 == "ctrl":
371 new_c = t2
372 else:
373 new_ttuple[idx2] = t2
374 # compute dst_cnt (modeB) from controller assignments in either participant
375 dst_cnt = cntv
376 if modeB:
377 for assign_text in (a1, a2):
378 cnt_eff = interpret_cnt_assignment(assign_text)
379 if cnt_eff:
380 typ, val = cnt_eff
381 if typ == "abs":
382 dst_cnt = val
383 else:
384 dst_cnt = (dst_cnt if dst_cnt is not None else 0) + val
385 if dst_cnt < 0:
386 dst_cnt = 0
387 if dst_cnt > N:
388 dst_cnt = N
389 # evaluate cnt-related conjuncts in guards (modeB) and build combined guard without cnt parts
390 if modeB:
391 allowed1, g1_label = sanitize_guard_for_modeB(g1, cntv, idx1 + 1 if comp1 == "train" else None)
392 allowed2, g2_label = sanitize_guard_for_modeB(g2, cntv, idx2 + 1 if comp2 == "train" else None)
393 if not (allowed1 and allowed2):
394 continue
395 guard_comb = None
396 if g1_label and g2_label:
397 guard_comb = f"{g1_label} and {g2_label}"
398 elif g1_label:
399 guard_comb = g1_label
400 elif g2_label:
401 guard_comb = g2_label
402 else:
403 # modeA: simply combine guards with train x replaced
404 g1r = replace_train_x(g1, idx1 + 1 if comp1 == "train" else None)
405 g2r = replace_train_x(g2, idx2 + 1 if comp2 == "train" else None)
406 parts = [p for p in (g1r, g2r) if p]
407 guard_comb = " and ".join(parts) if parts else None
408 # combine assignments: in modeB we remove cnt updates (normalize_assign does it)
409 if modeB:
410 assign_comb = join_assigns(normalize_assign(a1, idx1 + 1 if comp1 == "train" else None, modeB),
411 normalize_assign(a2, idx2 + 1 if comp2 == "train" else None, modeB))
412 else:
413 assign_comb = join_assigns(normalize_assign(a1, idx1 + 1 if comp1 == "train" else None, modeB),
414 normalize_assign(a2, idx2 + 1 if comp2 == "train" else None, modeB))
415 # include train resets
416 if comp1 == "train" and r1:
417 assign_comb = join_assigns(assign_comb, f"x{idx1+1} = 0")
418 if comp2 == "train" and r2:
419 assign_comb = join_assigns(assign_comb, f"x{idx2+1} = 0")
420 # build destination
421 dst = (new_g, new_c, tuple(new_ttuple), dst_cnt if modeB else None)
422 dst_idx = index[dst]
423 flat_trans.append({"from": s_idx, "to": dst_idx, "guard": guard_comb, "assign": assign_comb})
424
425 return states, flat_trans

430def write_xml(path: str, N: int, states: List[Tuple], transitions: List[Dict], modeB: bool):
431 lines: List[str] = []
432 lines.append('<?xml version="1.0" encoding="utf-8"?>')
433 lines.append('<!DOCTYPE nta PUBLIC "-//Uppaal Team//DTD Flat System 1.1//EN" '
434 '"http://www.it.uu.se/research/group/darts/uppaal/flat-1_1.dtd">')
435 lines.append('<nta>')
436 # Global declaration left minimal
437 lines.append(' <declaration>// All declarations are local to the template</declaration>')
438 # Template
439 lines.append(' <template>')
440 lines.append(' <name>Flatten</name>')
441 lines.append(' <declaration>')
442 # clocks: x1..xN, y, z
443 for i in range(N):
444 lines.append(f' clock x{i+1};')
445 lines.append(' clock y;')
446 lines.append(' clock z;')
447 # constants
448 for k, v in CONST.items():
449 lines.append(f' const int {k} = {v};')
450 # cnt only in mode A (kept as template-level int)
451 if not modeB:
452 lines.append(f' int {GLOBAL_INT_NAME};')
453 # channels
454 for ch in CHANNELS:
455 lines.append(f' chan {ch};')
456 lines.append(' </declaration>')
457
458 # locations
459 spacing = 220 # adjustable grid spacing
460 cols = int(max(1, round(len(states) ** 0.5)))
461 for idx, (gstate, cstate, ttuple, cntv) in enumerate(states):
462 x = spacing * (idx % cols)
463 y = spacing * (idx // cols)
464 base = f"{gstate}_{cstate}_" + "_".join(ttuple)
465 locname = f"{base}" + (f"_cnt{cntv}" if modeB else "")
466 # invariants
467 inv_parts = []
468 if gstate == "gate1":
469 inv_parts.append(f"y &lt;= {CONST['c']}")
470 if gstate == "gate3":
471 inv_parts.append(f"y &lt;= {CONST['d']}")
472 if cstate in ("controller1", "controller3"):
473 inv_parts.append(f"z &lt;= {CONST['e']}")
474 for i, t in enumerate(ttuple):
475 if t in ("train1", "train2", "train3"):
476 inv_parts.append(f"x{i+1} &lt;= {CONST['b']}")
477 lines.append(f' <location id="S{idx}" x="{x}" y="{y}">')
478 lines.append(f' <name>{locname}</name>')
479 if inv_parts:
480 lines.append(f' <label kind="invariant">{" and ".join(inv_parts)}</label>')
481 lines.append(' </location>')
482
483 # initial state
484 init_state = ("gate0", "controller0", tuple(["train0"] * N), 0 if modeB else None)
485 init_idx = states.index(init_state)
486 lines.append(f' <init ref="S{init_idx}"/>')
487
488 # transitions
489 for tr in transitions:
490 lines.append(' <transition>')
491 lines.append(f' <source ref="S{tr["from"]}"/>')
492 lines.append(f' <target ref="S{tr["to"]}"/>')
493 if tr.get("guard"):
494 lines.append(f' <label kind="guard">{tr["guard"]}</label>')
495 if tr.get("assign"):
496 lines.append(f' <label kind="assignment">{tr["assign"]}</label>')
497 lines.append(' </transition>')
498
499 lines.append(' </template>')
500
501 # system
502 lines.append(' <system>')
503 lines.append(' system Flatten;')
504 lines.append(' </system>')
505
506 # queries: 1) no two trains in train3, 2) gate3 and some train3
507 lines.append(' <queries>')
508 bad_states_1 = []
509 for (g, c, ttuple, cntv) in states:
510 cnt3 = sum(1 for t in ttuple if t == "train3")
511 if cnt3 >= 2:
512 base = f"{g}_{c}_" + "_".join(ttuple)
513 if modeB:
514 base += f"_cnt{cntv}"
515 bad_states_1.append(f"Flatten.{base}")
516 if bad_states_1:
517 formula = " or ".join(bad_states_1)
518 lines.append(' <query>')
519 lines.append(f' <formula>A[] !({formula})</formula>')
520 lines.append(' <comment>No two trains may be in train3 simultaneously</comment>')
521 lines.append(' </query>')
522
523 bad_states_2 = []
524 for (g, c, ttuple, cntv) in states:
525 if g == "gate3" and any(t == "train3" for t in ttuple):
526 base = f"{g}_{c}_" + "_".join(ttuple)
527 if modeB:
528 base += f"_cnt{cntv}"
529 bad_states_2.append(f"Flatten.{base}")
530 if bad_states_2:
531 formula = " or ".join(bad_states_2)
532 lines.append(' <query>')
533 lines.append(f' <formula>A[] !({formula})</formula>')
534 lines.append(' <comment>Gate cannot be up (gate3) while any train is in train3</comment>')
535 lines.append(' </query>')
536 lines.append(' </queries>')
537
538 lines.append('</nta>')
539
540 Path(path).write_text("\n".join(lines), encoding="utf-8")
541 print(f"Wrote flattened XML to: {path}")
542 print(f"States: {len(states)}, Transitions: {len(transitions)}")
