35def generate_iandpframes(num_frames: int) -> str:
36 """
37 Generate IandPframes process for num_frames BFrames.
38 - bstart extras: extra1 .. extra{N-1}
39 - ibx extras: extra{N} .. extra{2N-2}
40 - all extras urgent
41 - transitions comma-separated, last ends with semicolon
42 """
43 assert num_frames >= 1
44
45 # Each chain has (num_frames - 1) extras
46 num_extras = num_frames - 1
47 extras = [f"extra{i}" for i in range(1, 2 * num_extras + 1)]
48 b_extras = extras[:num_extras] # extra1 .. extra{N-1}
49 i_extras = extras[num_extras:] # extra{N} .. extra{2N-2}
50
51 # Locations: L0, all extras, L1..L8, I
52 locations = ", ".join(["L0"] + extras + ["L1", "L2", "L3", "L4", "L5", "L6", "L7", "L8", "I"])
53
54 transitions = []
55
56 # --- bstart chain: L1 -> extra1 -> ... -> extra{N-1} -> L0 ---
57 transitions.append(f"L1 -> {b_extras[0]} {{ guard x <= 0; sync bstart001!; }}")
58 for k in range(1, num_extras):
59 transitions.append(f"{b_extras[k-1]} -> {b_extras[k]} {{ guard x <= 0; sync bstart{k+1:03d}!; }}")
60 transitions.append(f"{b_extras[-1]} -> L0 {{ guard x <= 0; sync bstart{num_frames:03d}!; }}")
61
62 # --- fixed core transitions ---
63 transitions += [
64 "L2 -> L1 { guard x == 515; assign x = 0; }",
65 "L3 -> L2 { guard x == 521; sync dctq!; assign x = 0; }",
66 "L4 -> L3 { guard perror == 1 && x <= 0; assign x = 0, perror = 0; }",
67 "L5 -> L4 { guard x == 32; assign x = 0; }",
68 "L6 -> L5 { guard x >= 1180 && x <= 1181 && P == 1; assign mvp = 1, P = 0, x = 0; }",
69 ]
70
71 # --- ibx chain: L7 -> extraN -> ... -> extra{2N-2} -> L6 ---
72 transitions.append(f"L7 -> {i_extras[0]} {{ guard x <= 0; sync ibx001!; assign perror = 1, P = 1, x = 0; }}")
73 for k in range(1, num_extras):
74 transitions.append(f"{i_extras[k-1]} -> {i_extras[k]} {{ guard x <= 0; sync ibx{k+1:03d}!; assign perror = 1, P = 1, x = 0; }}")
75 transitions.append(f"{i_extras[-1]} -> L6 {{ guard x <= 0; sync ibx{num_frames:03d}!; assign perror = 1, P = 1, x = 0; }}")
76
77 # --- tail transitions ---
78 transitions += [
79 "L8 -> L7 { guard x == 673; assign x = 0; }",
80 "I -> L8 { guard x == 595; sync dctq1!; assign x = 0; }",
81 ]
82
83 # Format transitions: comma after every transition except last; indent 4 spaces
84 formatted_lines = []
85 for i, t in enumerate(transitions):
86 end = ";" if i == len(transitions) - 1 else ","
87 formatted_lines.append(f" {t}{end}")
88
89 urgent_decl = ", ".join(extras)
90
91 return f"""process IandPframes() {{
92 clock x;
93 int perror = 0;
94 int P = 0;
95 state {locations};
96 urgent {urgent_decl};
97 init I;
98 trans
99{chr(10).join(formatted_lines)}
100}}
101"""
