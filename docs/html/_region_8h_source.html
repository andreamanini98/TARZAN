357 struct RegionHash
358 {
359 std::size_t operator()(const Region &region) const
360 {
361 std::size_t seed = 0;
362
363 // Hash the location.
364 hash_combine(seed, region.q);
365
366 // Hash x0 bitset.
367 hash_combine(seed, hash_bitset(region.x0));
368
369 // Hash integer vector h.
370 boost::hash_range(seed, region.h.data(), region.h.data() + region.h.size());
371
372 // For deques, hash only size and first/last elements for speed.
373 const auto unbounded_size = region.unbounded.size();
374 hash_combine(seed, unbounded_size);
375 if (unbounded_size > 0)
376 {
377 hash_combine(seed, hash_bitset(region.unbounded.front()));
378 if (unbounded_size > 1)
379 hash_combine(seed, hash_bitset(region.unbounded.back()));
380 }
381
382 const auto bounded_size = region.bounded.size();
383 hash_combine(seed, bounded_size);
384 if (bounded_size > 0)
385 {
386 hash_combine(seed, hash_bitset(region.bounded.front()));
387 if (bounded_size > 1)
388 hash_combine(seed, hash_bitset(region.bounded.back()));
389 }
390
391 // Hash variables.
392 hash_combine(seed, region.variables.size());
393 // Only hash the first variable if present.
394 if (!region.variables.empty())
395 {
396 const auto &[varName, varValue] = *region.variables.begin();
397 hash_combine(seed, varName);
398 hash_combine(seed, varValue);
399 }
400
401 return seed;
402 }
403 };
