11def parse_table(table_text):
12 """Parse a single table and extract relevant information."""
13 # Extract table name from caption
14 caption_match = re.search(r'\\caption\{Benchmark results for (\w+)\}', table_text)
15 if not caption_match:
16 return None
17
18 table_name = caption_match.group(1)
19
20 # Extract data rows (skip header rows)
21 lines = table_text.split('\n')
22 data_rows = []
23
24 in_data = False
25 for line in lines:
26 # Skip until we hit the double hline (after header)
27 if '\\hline\\hline' in line:
28 in_data = True
29 continue
30
31 if not in_data:
32 continue
33
34 # Stop at the closing hline
35 if line.strip() == '\\hline':
36 break
37
38 # Skip empty lines
39 if not line.strip():
40 continue
41
42 # Parse data row
43 if '&' in line:
44 data_rows.append(line)
45
46 return {
47 'name': table_name,
48 'rows': data_rows

52def parse_row(row_text):
53 """Parse a single data row and extract values for each tool."""
54 # Remove trailing \\ and whitespace
55 row_text = row_text.strip().rstrip('\\\\').strip()
56
57 # Split by &
58 raw_parts = [p.strip() for p in row_text.split('&')]
59
60 # Expand multicolumn entries
61 parts = []
62 for part in raw_parts:
63 if '\\multicolumn' in part:
64 # Extract the span count and content from \multicolumn{N}{...}{content}
65 match = re.search(r'\\multicolumn\{(\d+)\}\{[^}]*\}\{([^}]*)\}', part)
66 if match:
67 span = int(match.group(1))
68 content = match.group(2)
69 # Add the content once, then fill with empty strings for the span
70 parts.append(content)
71 for _ in range(span - 1):
72 parts.append('')
73 else:
74 parts.append(part)
75 else:
76 parts.append(part)
77
78 # Need at least instance + 12 data columns (ignoring verification columns)
79 if len(parts) < 13:
80 return None
81
82 instance = parts[0]
83
84 # Extract N from instance name
85 # Handle formats like "flower_02", "fischer_02", "latch", etc.
86 instance_match = re.search(r'_0*(\d+)$', instance)
87 if instance_match:
88 n = instance_match.group(1)
89 else:
90 # For single instances without number (like "latch"), use "1"
91 # or extract any trailing number
92 num_match = re.search(r'\d+$', instance)
93 if num_match:
94 n = num_match.group(0)
95 else:
96 # Use instance name itself if no number found
97 n = instance.split('_')[-1] if '_' in instance else instance
98
99 # TARZAN: parts[1-4] (Time, Exec, Mem, Regions)
100 tarzan = {
101 'vt': parts[1],
102 'et': parts[2],
103 'mem': parts[3],
104 'regions': parts[4]
105 }
106
107 # TChecker: parts[5-8] (Time, Exec, Mem, States)
108 tchecker = {
109 'vt': parts[5],
110 'et': parts[6],
111 'mem': parts[7],
112 'states': parts[8]
113 }
114
115 # UPPAAL: parts[9-12] (Time, Exec, Mem, States)
116 uppaal = {
117 'vt': parts[9],
118 'et': parts[10],
119 'mem': parts[11],
120 'states': parts[12]
121 }
122
123 return {
124 'n': n,
125 'tarzan': tarzan,
126 'tchecker': tchecker,
127 'uppaal': uppaal
128 }

131def process_value(value, is_time=False):
132 """Process a value: handle KO, OOM, <0.001, etc."""
133 value = value.strip()
134
135 # Handle multicolumn KO
136 if '\\multicolumn' in value and 'KO' in value:
137 return 'KO'
138
139 # Handle direct KO or OOM
140 if value == 'KO' or value == 'OOM':
141 return value
142
143 # Handle < 0.001 - use left superscript downarrow
144 if value.startswith('$<$') or value.startswith('<'):
145 return '$^{\\downarrow}0.001$'
146
147 return value

150def format_tool_data(tool_data, tool_name):
151 """Format data for a single tool based on KO/OOM status."""
152 vt = process_value(tool_data['vt'], is_time=True)
153 et = process_value(tool_data['et'], is_time=True)
154 mem = process_value(tool_data['mem'])
155 count = process_value(tool_data.get('regions') or tool_data.get('states', ''))
156
157 # Check if any value is KO or if it's a multicolumn
158 is_ko = vt == 'KO' or et == 'KO' or mem == 'KO' or count == 'KO'
159 is_oom = 'OOM' in vt or 'OOM' in et or 'OOM' in mem or 'OOM' in count
160
161 # Handle multicolumn cases
162 if '\\multicolumn' in tool_data['vt']:
163 if tool_name == 'TARZAN':
164 # TARZAN: OOM in Mem, -- elsewhere
165 return '-- & -- & OOM & --'
166 else:
167 # TChecker/UPPAAL: left superscript uparrow with 600 in VT/ET, -- elsewhere
168 return '$^{\\uparrow}600$ & $^{\\uparrow}600$ & -- & --'
169
170 if is_ko or is_oom:
171 if tool_name == 'TARZAN':
172 # TARZAN: OOM in Mem, -- elsewhere
173 return '-- & -- & OOM & --'
174 else:
175 # TChecker/UPPAAL: left superscript uparrow with 600 in VT/ET, -- elsewhere
176 return '$^{\\uparrow}600$ & $^{\\uparrow}600$ & -- & --'
177
178 return f'{vt} & {et} & {mem} & {count}'

275def convert_tables(input_file, output_file=None):
276 """Convert tables from input file to output file."""
277 # Read input file
278 with open(input_file, 'r') as f:
279 content = f.read()
280
281 # Split into individual tables
282 table_pattern = r'\\begin{table}.*?\\end{table}'
283 table_matches = re.findall(table_pattern, content, re.DOTALL)
284
285 # Parse all tables
286 parsed_tables = []
287 for table_text in table_matches:
288 parsed = parse_table(table_text)
289 if parsed:
290 parsed_tables.append(parsed)
291
292 if not parsed_tables:
293 print("No tables found in input file")
294 return
295
296 # Create combined table
297 output_text = create_combined_table(parsed_tables)
298
299 # Write output
300 if output_file:
301 with open(output_file, 'w') as f:
302 f.write(output_text)
303 print(f"Converted {len(parsed_tables)} tables to {output_file}")
304 else:
305 print(output_text)
