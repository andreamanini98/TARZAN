20def generate_xta_model(num_keys: int) -> str:
21 lines = []
22 lines.append(f"// Auto-generated XTA model for {num_keys} keys\n")
23 lines.append("int gate;\n\n")
24
25 # --- key_i processes ---
26 for i in range(1, num_keys + 1):
27 lines.append(f"process key{i:02d}() {{")
28 lines.append(" clock x;")
29 locs = ",\n ".join(f"L{j}" for j in range(i + 1))
30 lines.append(" state")
31 lines.append(" " + locs + ";")
32 # init L0
33 lines.append(" init\n L0;")
34 lines.append(" trans")
35 trans = []
36 # first i-1 transitions with reset
37 for j in range(i - 1):
38 trans.append(f" L{j} -> L{j+1} {{ guard x == {i}; assign x = 0; }}")
39 # last transition increments gate
40 trans.append(f" L{i-1} -> L{i} {{ guard x == {i}; assign gate = gate + 1; }}")
41 lines.append(",\n".join(trans) + ";")
42 lines.append("}\n")
43
44 # --- unlocker() process ---
45 lines.append("process unlocker() {")
46 lines.append(" clock x;")
47 G_indices = [f"G{j}" for j in range(num_keys + 1)]
48 locations = G_indices + ["goal"]
49 lines.append(" state")
50 lines.append(" " + ",\n ".join(locations) + ";")
51 lines.append(" init\n G0;")
52 lines.append(" trans")
53 trans = []
54 # G0->G1..G{N-1}->G{N}
55 for j in range(num_keys):
56 trans.append(f" G{j} -> G{j+1} {{ guard x == {num_keys}; assign x = 0; }}")
57 # G{N}->goal guarded by gate == N
58 trans.append(f" G{num_keys} -> goal {{ guard x == {num_keys} && gate == {num_keys}; assign x = 0; }}")
59 lines.append(",\n".join(trans) + ";")
60 lines.append("}\n")

62 # --- system instantiation ---
63 instances = ", ".join([f"key{i:02d}" for i in range(1, num_keys + 1)] + ["unlocker"])
64 lines.append(f"system {instances};\n")
65
