22// The following is the grammar for the Liana DSL used to create Timed Automata.
23// Whether the actions are input or output actions must be specified only in the transitions.
24// T and F are syntax sugar for true and false.
25// Integer variables in Timed Automata are automatically initialized to 0. To overcome this, a transition can be added to initialize them in the Timed Automaton itself.
26//
27// <automaton> -> 'create' 'automaton' <literal> (eps | <symm_rule>)
28// '{'
29// 'clocks' '{' (eps | <literal> (, <literal>)* ';') '}'
30// 'actions' '{' <literal> (, <literal>)* ';' '}'
31// (eps | 'integers' '{' <literal> (, <literal>)* ';' '}')
32// 'locations' '{' <locations_rule> '}'
33// 'transitions' '{' <transition_rule> (, <transition_rule>)* ';' '}'
34// '}'
35//
36// <symm_rule> -> '::' 'symm' '<' <int> '>'
37//
38// <locations_rule> -> <loc_rule> (, <loc_rule>)* ';'
39//
40// <loc_rule> -> <literal> <loc_content_rule>
41//
42// <loc_content_rule> -> '<' (eps | <ini> | <urg> | <inv> | <ini> ',' <urg> | <ini> ',' <inv> | <urg> ',' <inv> | <ini> ',' <urg> ',' <inv>) '>'
43//
44// <ini> -> 'ini' ':' <bool>
45//
46// <urg> -> 'urg' ':' <bool>
47//
48// <inv> -> 'inv' ':' <guard_rule>
49//
50// <bool> -> 'T' | 'F' | 'true' | 'false'
51//
52// <transition_rule> -> '('
53// <literal> ','
54// <actions_rule> ','
55// <guard_rule> ','
56// (eps | <boolean_expr> ',')
57// '[' (eps | <literal> (',' <literal>)*) ']' ','
58// (eps | '[' <assignment_expr> (',' <assignment_expr>)* ']' ',')
59// <literal>
60// ')'
61//
62// <guard_rule> -> '[' (eps | <clock_constraint_rule> (, <clock_constraint_rule>)*) ']'
63//
64// <actions_rule> -> <literal> (eps | <input_output_action>)
65//
66// <input_output_action> -> '!' | '?'
67//
68// <clock_constraint_rule> -> '(' <literal> ',' <comparison_operator> ',' <int> ')'
69//
70// <comparison_operator> -> '<' | '<=' | '==' | '>=' | '>'
71//
72// <int> -> '1..9'('0..9')*
73//
74// <literal> -> ('a..z' | 'A..Z' | '0..9' | '_' )+
75//
76//
77// The following is the grammar of expressions.
78//
79// <assignment_expr> -> <variable> '=' <arithmetic_expr>
80//
81// <arithmetic_expr> -> <additive_expr>
82//
83// <additive_expr> -> <multiplicative_expr> (<add_op> <multiplicative_expr>)*
84//
85// <multiplicative_expr> -> <primary_expr> (<mult_op> <primary_expr>)*
86//
87// <primary_expr> -> <int> | <variable> | '(' <additive_expr> ')'
88//
89// <variable> -> <literal>
90//
91// <add_op> -> '+' | '-'
92//
93// <mult_op> -> '*' | '/'
94//
95// <boolean_expr> -> <boolean_or_expr>
96//
97// <boolean_or_expr> -> <boolean_and_expr> (<or_op> <boolean_and_expr>)*
98//
99// <boolean_and_expr> -> <boolean_term> (<and_op> <boolean_term>)*
100//
101// <boolean_term> -> '(' <boolean_or_expr> ')' | <comparison_expr> | <bool>
102//
103// <comparison_expr> <arithmetic_expr> <comparison_operator> <arithmetic_expr>
104//
105// <or_op> -> '||'
106//
107// <and_op> -> '&&'
108
