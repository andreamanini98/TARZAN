33def escape_uppaal(s: str) -> str:
34 if not s:
35 return s
36 # order matters: handle <= >= before < >
37 s = s.replace(">=", "&gt;=")
38 s = s.replace("<=", "&lt;=")
39 s = s.replace(">", "&gt;")
40 s = s.replace("<", "&lt;")
41 return s

43def split_conjuncts(guard: str):
44 """Split guard into conjuncts using 'and' and '&&' heuristically."""
45 if not guard:
46 return []
47 g = guard.replace("&&", " and ")
48 parts = [p.strip() for p in g.split(" and ") if p.strip()]
49 return parts

51def id_condition_holds(cond: str, proc_index: int, current_id: int) -> bool:
52 """
53 Evaluate id-related condition cond at runtime:
54 - cond examples: 'id == 0', 'id == pid', 'id != 0', 'id != pid'
55 Returns True if condition holds for (proc_index (0-based), current_id).
56 If condition is not recognized, conservatively return False.
57 """
58 if cond is None or cond.strip() == "":
59 return True
60 c = cond.replace(" ", "")
61 if "id==" in c:
62 rhs = c.split("id==",1)[1]
63 if rhs == "pid":
64 return current_id == (proc_index + 1)
65 try:
66 return current_id == int(rhs)
67 except ValueError:
68 return False
69 if "id!=" in c:
70 rhs = c.split("id!=",1)[1]
71 if rhs == "pid":
72 return current_id != (proc_index + 1)
73 try:
74 return current_id != int(rhs)
75 except ValueError:
76 return True
77 # unknown id expression -> conservative False
78 return False

80def remove_id_conjuncts(parts, proc_index, current_id):
81 """
82 Given conjunct parts (list of strings), evaluate id-related conjuncts against current_id.
83 Return (allowed: bool, remaining_parts: list[str]) where remaining_parts excludes id conjuncts.
84 """
85 remaining = []
86 for p in parts:
87 if "id" in p:
88 if not id_condition_holds(p, proc_index, current_id):
89 return False, []
90 # if holds, we drop the id conjunct from remaining (id encoded in location name)
91 else:
92 remaining.append(p)
93 return True, remaining

95def normalize_assignments(assign_text: str, proc_index: int):
96 """
97 Normalize assignment text:
98 - remove id assignments entirely (they only affect destination suffix)
99 - replace 'pid' with proc_index+1
100 - replace 'x' with x{proc_index+1}
101 - split separators ';' or ',' and rejoin with ', ' for UPPAAL
102 Returns list of assignment strings (may be empty).
103 """
104 if not assign_text:
105 return []
106 parts = [p.strip() for p in assign_text.replace(";", ",").split(",") if p.strip()]
107 out = []
108 for a in parts:
109 a_nospace = a.replace(" ", "")
110 if a_nospace.startswith("id="):
111 # drop id assignments entirely
112 continue
113 # replace pid and x occurrences (simple replace)
114 a2 = a.replace("pid", str(proc_index+1)).replace("x", f"x{proc_index+1}")
115 out.append(a2)
116 # remove duplicates while preserving order
117 seen = set()
118 dedup = []
119 for a in out:
120 key = a.replace(" ", "")
121 if key not in seen:
122 seen.add(key)
123 dedup.append(a)
124 return dedup

129def build_states(N):
130 product_locs = list(itertools.product(LOCATIONS, repeat=N)) # tuples of location names
131 states = []
132 for loc_tuple in product_locs:
133 for idv in range(0, N+1):
134 states.append((loc_tuple, idv))
135 return product_locs, states

140def generate_transitions(N, product_locs, states):
141 state_index = {s: i for i, s in enumerate(states)}
142 transitions = []
143
144 for s_idx, (loc_tuple, idv) in enumerate(states):
145 for p in range(N): # process p moves
146 local = loc_tuple[p]
147 for (src, tgt, guard_text, assign_text, resets_x) in SINGLE_TRANSITIONS:
148 if src != local:
149 continue
150
151 # split conjuncts and evaluate id-related conjuncts against idv
152 conjuncts = split_conjuncts(guard_text)
153 ok, remaining = remove_id_conjuncts(conjuncts, p, idv)
154 if not ok:
155 continue # id-related guard fails => transition not allowed from this state
156
157 # build final guard from remaining conjuncts:
158 final_guard_parts = []
159 for r in remaining:
160 # replace pid and x with per-process ones; escape
161 r2 = r.replace("pid", str(p+1)).replace("x", f"x{p+1}")
162 r2 = escape_uppaal(r2)
163 if r2:
164 final_guard_parts.append(r2)
165 final_guard = " and ".join(final_guard_parts) if final_guard_parts else None
166
167 # determine destination id suffix based on assignments in assign_text
168 dest_id = idv
169 if assign_text:
170 # check id assignments in original assign_text
171 parts = [a.strip() for a in assign_text.replace(";", ",").split(",") if a.strip()]
172 for a in parts:
173 a_ns = a.replace(" ", "")
174 if a_ns.startswith("id="):
175 rhs = a_ns.split("=",1)[1]
176 if rhs == "pid":
177 dest_id = p+1
178 else:
179 try:
180 dest_id = int(rhs)
181 except ValueError:
182 pass
183
184 # Build destination location tuple
185 dst_tuple = list(loc_tuple)
186 dst_tuple[p] = tgt
187 dst_state = (tuple(dst_tuple), dest_id)
188 if dst_state not in state_index:
189 # shouldn't happen (we built full state space)
190 continue
191 dst_idx = state_index[dst_state]
192
193 # Build final assignments excluding id=... (they are encoded in dest_state)
194 assigns = normalize_assignments(assign_text, p)
195 # include clock reset if needed (ensure no duplicate)
196 clock_assign = f"x{p+1} = 0"
197 if resets_x and not any(a.replace(" ", "") == clock_assign.replace(" ", "") for a in assigns):
198 assigns.append(clock_assign)
199 final_assign = ", ".join(assigns) if assigns else None
200
201 transitions.append({
202 "from": s_idx,
203 "to": dst_idx,
204 "guard": final_guard,
205 "assign": final_assign,
206 "comment": f"proc{p+1}:{src}->{tgt} (id {idv} -> {dest_id})"
207 })
208 return transitions

213def build_mutual_query(product_locs, N):
214 bad = []
215 for loc_tuple in product_locs:
216 cs_count = sum(1 for l in loc_tuple if l == "cs")
217 if cs_count >= 2:
218 base = "_".join(f"{l}{i+1}" for i, l in enumerate(loc_tuple))
219 for idv in range(0, N+1):
220 name = f"Fischer_flat.{base}_id{idv}"
221 bad.append(name)
222 if not bad:
223 return None
224 formula = " or ".join(bad)
225 # E<> (bad) would be reachability of bad — you asked earlier for a query that says we cannot reach any location with 2 processes in cs.
226 # We emit a property expressing the negation as A[] !(bad), but keep an explicit reachability query as well.
227 lines = []
228 # Reachability query (E<> bad): if reachable, mutual exclusion violated
229 lines.append(" <query>")
230 lines.append(f" <formula>E&lt;&gt; ({formula})</formula>")
231 lines.append(" <comment>Mutual exclusion violation: at least 2 processes in cs</comment>")
232 lines.append(" </query>")
233 # Also add the invariance form A[] !(bad) as a separate query
234 lines.append(" <query>")
235 lines.append(f" <formula>A[] !({formula})</formula>")
236 lines.append(" <comment>Mutual exclusion holds: never two processes in cs</comment>")
237 lines.append(" </query>")
238 return "\n".join(lines)

269 for idx, (loc_tuple, idv) in enumerate(states):
270 x = spacing * (idx % cols)
271 y = spacing * (idx // cols)
272 base = "_".join(f"{l}{i+1}" for i, l in enumerate(loc_tuple))
273 locname = f"{base}_id{idv}"

283 lines.append(f' <label kind="comment">{base}_id{idv}</label>')
