182 struct NetworkRegionHash
183 {
184 std::size_t operator()(const NetworkRegion &networkRegion) const
185 {
186 std::size_t seed = 0;
187
188 // Hashing regions.
189 constexpr region::RegionHash regionHash;
190 for (const auto &region: networkRegion.getRegions())
191 hash_combine(seed, regionHash(region));
192
193 // Hashing isAorC.
194 // The set must preserve the order of its elements for the hash to work.
195 hash_combine(seed, networkRegion.isAorC.size());
196 for (int val: networkRegion.isAorC)
197 hash_combine(seed, val);
198
199 // Hashing clockOrdering.
200 hash_combine(seed, networkRegion.clockOrdering.size());
201 for (const auto &map: networkRegion.clockOrdering)
202 {
203 // The map must preserve the order of its elements for the hash to work.
204 hash_combine(seed, map.size());
205 for (const auto &[key, bitset]: map)
206 {
207 hash_combine(seed, key);
208 hash_combine(seed, hash_bitset(bitset));
209 }
210 }
211
212 // Hashing the integer variables.
213 hash_combine(seed, networkRegion.networkVariables.size());
214 for (const auto &[varName, varValue]: networkRegion.networkVariables)
215 {
216 hash_combine(seed, varName);
217 hash_combine(seed, varValue);
218 }
219
220 return seed;
221 }
222 };
