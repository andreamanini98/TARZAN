20 def __init__(self, content: str):
21 self.content = content
22 self.constants: Dict[str, str] = {}
23 self.global_vars: List[Tuple[str, str]] = [] # (type, name)
24 self.global_clocks: List[str] = [] # Global clock names
25 self.channels: List[str] = []
26 self.processes: List[Dict] = []

28 def parse(self):
29 """Parse the XTA file content."""
30 # Remove comments
31 content = re.sub(r'//.*?\n', '\n', self.content)
32 content = re.sub(r'/\*.*?\*/', '', content, flags=re.DOTALL)
33
34 # Parse constants
35 for match in re.finditer(r'const\s+int\s+(\w+)\s*=\s*([^;]+);', content):
36 name, value = match.groups()
37 self.constants[name] = value.strip()
38
39 # Parse global clocks (clocks declared outside process blocks)
40 # First, find all process blocks to exclude them
41 process_blocks = []
42 for match in re.finditer(r'process\s+\w+\s*\‍([^)]*\‍)\s*\{.*?\n\}', content, re.DOTALL):
43 process_blocks.append((match.start(), match.end()))
44
45 # Find global clock declarations (outside processes)
46 for match in re.finditer(r'^clock\s+([^;]+);', content, re.MULTILINE):
47 # Check if this clock declaration is outside all process blocks
48 pos = match.start()
49 is_global = True
50 for start, end in process_blocks:
51 if start <= pos <= end:
52 is_global = False
53 break
54
55 if is_global:
56 # Parse comma-separated clock names
57 clock_list = match.group(1)
58 clocks = [c.strip() for c in clock_list.split(',')]
59 self.global_clocks.extend(clocks)
60
61 # Parse global variables
62 for match in re.finditer(r'^(bool|int)\s+(\w+)\s*;', content, re.MULTILINE):
63 var_type, var_name = match.groups()
64 self.global_vars.append((var_type, var_name))
65
66 # Parse channels
67 for match in re.finditer(r'chan\s+([^;]+);', content):
68 channels = [ch.strip() for ch in match.group(1).split(',')]
69 self.channels.extend(channels)
70
71 # Parse processes
72 process_pattern = r'process\s+(\w+)\s*\‍([^)]*\‍)\s*\{(.*?)\n\}'
73 for match in re.finditer(process_pattern, content, re.DOTALL):
74 process_name = match.group(1)
75 process_body = match.group(2)
76 process_data = self.parse_process(process_name, process_body)
77 if process_data:
78 self.processes.append(process_data)
79
80 # Parse system declaration
81 system_match = re.search(r'system\s+([^;]+);', content)
82 if system_match:
83 self.system_processes = [p.strip() for p in system_match.group(1).split(',')]
84 else:
85 self.system_processes = [p['name'] for p in self.processes]

87 def parse_process(self, name: str, body: str) -> Dict:
88 """Parse a single process definition."""
89 process = {
90 'name': name,
91 'clocks': [],
92 'states': {},
93 'urgent_states': set(),
94 'init_state': None,
95 'transitions': []
96 }
97
98 # Parse local clocks (can be comma-separated)
99 for match in re.finditer(r'clock\s+([^;]+);', body):
100 clock_list = match.group(1)
101 clocks = [c.strip() for c in clock_list.split(',')]
102 process['clocks'].extend(clocks)
103
104 # Parse urgent locations
105 urgent_section = re.search(r'urgent\s+(.*?)(?:init|trans|state)', body, re.DOTALL)
106 if urgent_section:
107 urgent_text = urgent_section.group(1)
108 # Parse comma-separated urgent location names
109 for line in urgent_text.split(','):
110 loc = line.strip().rstrip(';')
111 if loc:
112 process['urgent_states'].add(loc)
113
114 # Parse states with invariants
115 state_section = re.search(r'state\s+(.*?)(?:\burgent\b|\binit\b|\btrans\b)', body, re.DOTALL)
116 if state_section:
117 state_text = state_section.group(1)
118 # Remove any semicolons at the end
119 state_text = state_text.rstrip(';').strip()
120
121 # Parse state declarations - they are comma-separated
122 # Each state can be: "name" or "name { invariant }"
123 states_list = []
124 current_state = ""
125 brace_count = 0
126
127 for char in state_text:
128 if char == '{':
129 brace_count += 1
130 current_state += char
131 elif char == '}':
132 brace_count -= 1
133 current_state += char
134 elif char == ',' and brace_count == 0:
135 # End of a state declaration
136 states_list.append(current_state.strip())
137 current_state = ""
138 else:
139 current_state += char
140
141 # Don't forget the last state
142 if current_state.strip():
143 states_list.append(current_state.strip())
144
145 # Now parse each state
146 for state_decl in states_list:
147 if not state_decl:
148 continue
149
150 # Check for invariant
151 inv_match = re.match(r'(\w+)\s*\{([^}]+)\}', state_decl)
152 if inv_match:
153 state_name = inv_match.group(1)
154 invariant = inv_match.group(2).strip()
155 process['states'][state_name] = {'invariant': invariant}
156 else:
157 # Remove any trailing semicolons and whitespace from state name
158 state_name = state_decl.strip().rstrip(';').strip()
159 if state_name:
160 process['states'][state_name] = {'invariant': None}
161
162 # Parse init state
163 init_match = re.search(r'init\s+(\w+);', body)
164 if init_match:
165 process['init_state'] = init_match.group(1)
166
167 # Parse transitions
168 trans_section = re.search(r'trans\s+(.*)', body, re.DOTALL)
169 if trans_section:
170 trans_text = trans_section.group(1)
171 # Split by transitions (semicolon separated)
172 transitions = re.findall(r'(\w+)\s*->\s*(\w+)\s*\{([^}]*)\}', trans_text)
173 for source, target, attrs in transitions:
174 trans = {
175 'source': source,
176 'target': target,
177 'guard': None,
178 'sync': None,
179 'sync_type': None,
180 'assign': None
181 }
182
183 # Parse guard
184 guard_match = re.search(r'guard\s+([^;]+);', attrs)
185 if guard_match:
186 trans['guard'] = guard_match.group(1).strip()
187
188 # Parse sync
189 sync_match = re.search(r'sync\s+(\w+)([!?])\s*;', attrs)
190 if sync_match:
191 trans['sync'] = sync_match.group(1)
192 trans['sync_type'] = sync_match.group(2)
193
194 # Parse assign
195 assign_match = re.search(r'assign\s+([^;]+);', attrs)
196 if assign_match:
197 trans['assign'] = assign_match.group(1).strip()
198
199 process['transitions'].append(trans)
200
201 return process

213class TCKGenerator:
214 """Generator for TChecker TCK format."""
215
216 def __init__(self, parser: XTAParser, system_name: str):
217 self.parser = parser
218 self.system_name = system_name
219 self.events: Set[str] = set()
220 self.tau_event = 'tau' # Default event for unsynchronized transitions
221
222 def generate(self) -> str:
223 """Generate TCK format output."""
224 lines = []
225
226 # System declaration (must be first)
227 lines.append(f"system:{self.system_name}\n\n")
228
229 # Collect all events from transitions
230 self.collect_events()
231
232 # Add tau event for unsynchronized transitions
233 if self.tau_event:
234 self.events.add(self.tau_event)
235
236 # Event declarations
237 lines.append("# Events\n")
238 for event in sorted(self.events):
239 lines.append(f"event:{event}\n")
240 lines.append("\n\n\n")
241
242 # Global clock and variable declarations
243 global_decls = self.generate_global_declarations()
244 if global_decls:
245 lines.append("# Global declarations\n")
246 lines.extend(global_decls)
247 lines.append("\n\n\n")
248
249 # Process declarations
250 lines.append("# Processes\n")
251 for process in self.parser.processes:
252 lines.append(f"process:{process['name']}\n")
253 lines.append("\n")
254
255 # Process-specific declarations (clocks, locations, edges)
256 for i, process in enumerate(self.parser.processes):
257 if i > 0:
258 lines.append("\n\n\n") # Three blank lines between processes (total spacing)
259
260 lines.append(f"# Process: {process['name']}\n")
261
262 # Local clocks
263 if process['clocks']:
264 lines.append(f"## Clocks for {process['name']}\n")
265 for clock in process['clocks']:
266 lines.append(f"clock:1:{process['name']}_{clock}\n")
267 lines.append("\n")
268
269 # Locations
270 locations = self.generate_locations(process)
271 if locations:
272 lines.append(f"## Locations for {process['name']}\n")
273 lines.extend(locations)
274 lines.append("\n")
275
276 # Edges
277 edges = self.generate_edges(process)
278 if edges:
279 lines.append(f"## Edges for {process['name']}\n")
280 lines.extend(edges)
281
282 # Synchronization constraints
283 sync_constraints = self.generate_sync_constraints()
284 if sync_constraints:
285 lines.append("\n\n\n# Synchronizations\n")
286 lines.extend(sync_constraints)
287
288 return ''.join(lines)

290 def collect_events(self):
291 """Collect all unique events from process transitions."""
292 for process in self.parser.processes:
293 for trans in process['transitions']:
294 if trans['sync']:
295 self.events.add(trans['sync'])

297 def generate_global_declarations(self) -> List[str]:
298 """Generate global clock and variable declarations."""
299 lines = []
300
301 # Constants (const int) - declare as initialized integer variables
302 for const_name, const_value in self.parser.constants.items():
303 # Parse the constant value (should be an integer)
304 try:
305 init_value = int(const_value)
306 # Use a reasonable range: allow the value itself and some flexibility
307 # For TChecker, we use min=0 and max as a reasonable upper bound
308 max_value = max(1000, init_value * 2)
309 lines.append(f"int:1:0:{max_value}:{init_value}:{const_name}\n")
310 except ValueError:
311 # If it's not a simple integer, use default range
312 lines.append(f"int:1:0:1000:0:{const_name}\n")
313
314 # Global clocks
315 for clock_name in self.parser.global_clocks:
316 lines.append(f"clock:1:{clock_name}\n")
317
318 # Global variables
319 for var_type, var_name in self.parser.global_vars:
320 if var_type == 'bool':
321 # bool in XTA is 0 or 1
322 lines.append(f"int:1:0:1:0:{var_name}\n")
323 elif var_type == 'int':
324 # Default int range (can be adjusted based on model)
325 lines.append(f"int:1:0:1000:0:{var_name}\n")
326
327 return lines

329 def generate_locations(self, process: Dict) -> List[str]:
330 """Generate location declarations for a process."""
331 lines = []
332
333 for state_name, state_data in process['states'].items():
334 attrs = []
335
336 # Check if initial location
337 if state_name == process['init_state']:
338 attrs.append("initial:")
339
340 # Check if urgent location
341 if state_name in process['urgent_states']:
342 attrs.append("urgent:")
343
344 # Add invariant if present
345 if state_data['invariant']:
346 invariant = self.convert_expression(state_data['invariant'], process)
347 attrs.append(f"invariant:{invariant}")
348
349 # Format location declaration
350 # Use " : " (space-colon-space) as separator between attributes
351 # If there are attributes, use {}, otherwise no braces
352 if attrs:
353 attr_str = "{" + " : ".join(attrs) + "}"
354 else:
355 attr_str = ""
356
357 lines.append(f"location:{process['name']}:{state_name}{attr_str}\n")
358
359 return lines

361 def generate_edges(self, process: Dict) -> List[str]:
362 """Generate edge declarations for a process."""
363 lines = []
364
365 for trans in process['transitions']:
366 source = trans['source']
367 target = trans['target']
368
369 # Determine event for this edge
370 event = trans['sync'] if trans['sync'] else self.tau_event
371
372 attrs = []
373
374 # Add guard (provided)
375 if trans['guard']:
376 guard = self.convert_expression(trans['guard'], process)
377 attrs.append(f"provided:{guard}")
378
379 # Add assignment (do)
380 if trans['assign']:
381 assign = self.convert_assignment(trans['assign'], process)
382 attrs.append(f"do:{assign}")
383
384 # Format edge declaration
385 # Use " : " (space-colon-space) as separator between attributes
386 attr_str = "{" + " : ".join(attrs) + "}" if attrs else ""
387 lines.append(f"edge:{process['name']}:{source}:{target}:{event}{attr_str}\n")
388
389 return lines

391 def generate_sync_constraints(self) -> List[str]:
392 """Generate synchronization constraints based on channels."""
393 lines = []
394
395 # Build maps of senders (!) and receivers (?) for each event
396 senders: Dict[str, List[str]] = {} # event -> list of process names with !
397 receivers: Dict[str, List[str]] = {} # event -> list of process names with ?
398
399 for process in self.parser.processes:
400 for trans in process['transitions']:
401 if trans['sync']:
402 event = trans['sync']
403 sync_type = trans['sync_type']
404
405 if sync_type == '!':
406 if event not in senders:
407 senders[event] = []
408 senders[event].append(process['name'])
409 elif sync_type == '?':
410 if event not in receivers:
411 receivers[event] = []
412 receivers[event].append(process['name'])
413
414 # Create pairwise synchronization constraints
415 # Each sender (!) pairs with each receiver (?) for the same event
416 all_events = set(senders.keys()) | set(receivers.keys())
417
418 for event in sorted(all_events):
419 event_senders = sorted(set(senders.get(event, [])))
420 event_receivers = sorted(set(receivers.get(event, [])))
421
422 if not event_senders or not event_receivers:
423 # Skip if no sender or no receiver for this event
424 continue
425
426 # Create pairwise synchronizations
427 # Each sender synchronizes with each receiver
428 for sender in event_senders:
429 for receiver in event_receivers:
430 sync_constraint = f"{sender}@{event}:{receiver}@{event}"
431 lines.append(f"sync:{sync_constraint}\n")
432
433 return lines

435 def convert_expression(self, expr: str, process: Dict) -> str:
436 """Convert XTA expression to TChecker expression."""
437 # Do NOT substitute constants - they are now declared as variables
438 result = expr
439
440 # Replace local clock references with process-prefixed names
441 # (but not global clocks)
442 for clock in process['clocks']:
443 if clock not in self.parser.global_clocks:
444 result = re.sub(r'\b' + clock + r'\b', f"{process['name']}_{clock}", result)
445
446 # Convert 'and' keyword to '&&' operator
447 result = re.sub(r'\band\b', '&&', result)
448
449 # Convert 'or' keyword to '||' operator
450 result = re.sub(r'\bor\b', '||', result)
451
452 # Add spaces around operators for better readability
453 # First, normalize any existing spacing around operators
454 result = re.sub(r'\s*([+\-*/=!<>]+)\s*', r' \1 ', result)
455
456 # Clean up multiple spaces
457 result = re.sub(r'\s+', ' ', result)
458
459 # Note: Arithmetic operators (+, -, *, /) are already compatible with TChecker
460 # Note: Comparison operators (==, !=, <, <=, >, >=) are already compatible
461
462 return result.strip()

464 def convert_assignment(self, assign: str, process: Dict) -> str:
465 """Convert XTA assignment to TChecker assignment."""
466 # Do NOT substitute constants - they are now declared as variables
467 result = assign
468
469 # Replace local clock references with process-prefixed names
470 # (but not global clocks)
471 for clock in process['clocks']:
472 if clock not in self.parser.global_clocks:
473 result = re.sub(r'\b' + clock + r'\b', f"{process['name']}_{clock}", result)
474
475 # Normalize assignment operator to =
476 result = result.replace(':=', '=')
477
478 # Handle multiple assignments (comma-separated)
479 assignments = [a.strip() for a in result.split(',')]
480 return "; ".join(assignments)

483def convert_xta_to_tck(xta_path: str, tck_path: str):
484 """Convert a single XTA file to TCK format."""
485 # Read XTA file
486 with open(xta_path, 'r') as f:
487 xta_content = f.read()
488
489 # Parse XTA
490 parser = XTAParser(xta_content)
491 parser.parse()
492
493 # Generate system name from filename
494 system_name = Path(xta_path).stem
495
496 # Generate TCK
497 generator = TCKGenerator(parser, system_name)
498 tck_content = generator.generate()
499
500 # Write TCK file
501 os.makedirs(os.path.dirname(tck_path), exist_ok=True)
502 with open(tck_path, 'w') as f:
503 f.write(tck_content)
504
505 print(f"Converted: {xta_path} -> {tck_path}")
