95def generate_xta(n: int) -> str:
96 """Generate the full XTA source for n patients."""
97 # --- Global channels ---
98 chan_lines = [f" patientArrivalDoc{i:03d}, patientArrivalNurse{i:03d}," for i in range(1, n + 1)]
99 chan_block = HEADER + "\n".join(chan_lines)[:-1] + ";\n\n" + CONSTANTS
100
101 # --- Patients ---
102 patients_block = "\n".join(PATIENT_TEMPLATE.format(idx=f"{i:03d}") for i in range(1, n + 1))
103
104 # --- Doctor transitions ---
105 doc_trans = ",\n".join(
106 f" Idle -> GoToPatient {{ sync patientArrivalDoc{i:03d}?; assign x = 0, docReady = 0; }}"
107 for i in range(1, n + 1)
108 )
109 doctor_block = DOCTOR_TEMPLATE.format(transitions=doc_trans)
110
111 # --- Nurse transitions ---
112 nurse_trans = ",\n".join(
113 f" Idle -> GoToPatient {{ sync patientArrivalNurse{i:03d}?; assign x = 0, nurseReady = 0; }}"
114 for i in range(1, n + 1)
115 )
116 nurse_block = NURSE_TEMPLATE.format(transitions=nurse_trans)
117
118 # --- System composition ---
119 system_block = "system " + ", ".join(
120 [f"Patient{i:03d}" for i in range(1, n + 1)] + ["Nurse", "Doctor"]
121 ) + ";"
122
123 return "\n".join([chan_block, patients_block, nurse_block, doctor_block, system_block])

126def generate_query(n: int) -> str:
127 """Generate a reachability query checking that all patients reach Done."""
128 conjuncts = " and ".join([f"Patient{i:03d}.Done" for i in range(1, n + 1)])
129 return f"E<> ( {conjuncts} )\n"
