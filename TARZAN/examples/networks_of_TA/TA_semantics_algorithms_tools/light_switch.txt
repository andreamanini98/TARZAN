create automaton light_switch
{
    clocks
    {
        x;
    }

    actions
    {
        press;
    }

    locations
    {
        off    <ini: true>,
        dim    <ini: false>,
        bright <ini: false>;
    }

    transitions
    {
        (off,    press?, [(x, >=, 0)],  [x], dim),
        (dim,    press?, [(x, >, 10)],  [],  off),
        (dim,    press?, [(x, <=, 10)], [],  bright),
        (bright, press?, [(x, >=, 0)],  [],  off);
    }
}
