create automaton fischer01
{
	clocks
	{
		x;
	}

	actions
	{
		a;
	}

	integers
	{
	    id;
	}

	locations
	{
		A    <ini: T>,
		req  <ini: F, inv: [(x, <=, 2)]>,
		wait <ini: F>,
		cs   <ini: F>;
	}

	transitions
	{
		(A,    a, [],           id == 0, [x],           req),
		(req,  a, [(x, <=, 2)],          [x], [id = 2], wait),
		(wait, a, [],           id == 0, [x],           req),
		(wait, a, [(x, >, 2)],  id == 2, [],            cs),
		(cs,   a, [],                    [],  [id = 0], A);
	}
}
