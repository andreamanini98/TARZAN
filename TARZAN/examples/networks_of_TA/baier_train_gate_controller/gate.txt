create automaton gate
{
    clocks
    {
        x;
    }

    actions
    {
        lower, raise, tau;
    }

    locations
    {
        up          <ini: true>,
        coming_down <ini: false>,
        down        <ini: false>,
        going_up    <ini: false>;
    }

    transitions
    {
        (up,          lower!, [(x, <=, 1)],             [x], coming_down),
        (coming_down, tau,    [(x, <=, 1)],             [],  down),
        (down,        raise!, [(x, <=, 2)],             [x], going_up),
        (going_up,    tau,    [(x, >=, 1), (x, <=, 2)], [],  up);
    }
}
