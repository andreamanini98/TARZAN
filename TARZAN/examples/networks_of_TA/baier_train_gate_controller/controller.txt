create automaton controller
{
    clocks
    {
        z;
    }

    actions
    {
        approach, lower, exit, raise;
    }

    locations
    {
        0 <ini: T>,
        1 <ini: F>,
        2 <ini: F>,
        3 <ini: F>;
    }

    transitions
    {
        (0, approach?, [(z, <=, 1)], [z], 1),
        (1, lower?,    [(z, <=, 1), (z, =, 1)],  [],  2),
        (2, exit?,     [(z, <=, 1)], [z], 3),
        (3, raise?,    [(z, <=, 1)], [],  0);
    }
}
