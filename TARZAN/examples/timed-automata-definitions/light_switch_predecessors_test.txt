create automaton light_switch
{
    clocks
    {
        x, y;
    }

    actions
    {
        switch_on, switch_off;
    }

    locations
    {
        off <initial: true>,
        on  <initial: false>;
    }

    transitions
    {
        (on, switch_off, [(y, >=, 2), (x, <, 4)], [x, y], off);
    }
}
