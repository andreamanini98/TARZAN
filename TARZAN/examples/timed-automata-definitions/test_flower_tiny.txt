create automaton test_flower
{
    clocks
    {
        x1, x2, x3, x4, y;
    }

    actions
    {
        a;
    }

    locations
    {
        start <ini: T>,
        goal  <ini: F>;
    }

    transitions
    {
        (start, a, [(x1, ==, 1)],   [x1],  start),
        (start, a, [(x2, ==, 2)],   [x2],  start),
        (start, a, [(x3, ==, 3)],   [x3],  start),
        (start, a, [(x4, ==, 4)],   [x4],  start),
        (start, a, [(x1, ==, 0),
                    (x2, ==, 0),
                    (x3, ==, 0),
                    (x4, ==, 0),
                    (y, >=, 1),
                    (y, <=, 12)], [], goal);
    }
}
