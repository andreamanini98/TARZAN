create automaton fischer
{
	clocks
	{
		x;
	}

	actions
	{
		a;
	}

	locations
	{
		A    <ini: T>,
		req  <>,
		wait <>,
		cs   <>;
	}

	transitions
	{
		(A,    a, [(x, >=, 0)], [x], req),
		(req,  a, [(x, <=, 2)], [x], wait),
		(wait, a, [(x, >=, 0)], [x], req),
		(wait, a, [(x, >, 2)],  [],  cs),
		(A,    a, [(x, >=, 0)], [],  A);
	}
}
