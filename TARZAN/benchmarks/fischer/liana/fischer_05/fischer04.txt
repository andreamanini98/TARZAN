create automaton fischer04
{
    clocks { x; }
    actions { a; }
    integers { id; }
    locations {
        wait <ini: F                   >,
        req  <ini: F, inv: [(x, <=, 2)]>,
        A    <ini: T                   >,
        cs   <ini: F                   >;
    }
    transitions {
        (A, a, [], id == 0, [x], req),
        (req, a, [(x, <=, 2)], [x], [id = 4], wait),
        (wait, a, [], id == 0, [x], req),
        (wait, a, [(x, >, 2)], id == 4, [], cs),
        (cs, a, [], [], [id = 0], A);
    }
}
