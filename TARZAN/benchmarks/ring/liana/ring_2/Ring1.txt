create automaton Ring1
{
    clocks { x; }
    actions { a; }
    locations {
        L0   <ini: T>,
        L1   <ini: F>,
        L2   <ini: F>,
        L3   <ini: F>,
        L4   <ini: F>,
        goal <ini: F>;
    }
    transitions {
        (goal, a, [(x, ==, 2)], [x], L0),
        (L4, a, [(x, ==, 2)], [x], goal),
        (L3, a, [(x, ==, 2)], [x], L4),
        (L2, a, [(x, ==, 2)], [x], L3),
        (L1, a, [(x, ==, 2)], [x], L2),
        (L0, a, [(x, ==, 2)], [x], L1);
    }
}
