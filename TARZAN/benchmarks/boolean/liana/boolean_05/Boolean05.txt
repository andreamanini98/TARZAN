create automaton Boolean05
{
    clocks { x; }
    actions { a; }
    integers { ctr1, ctr2, ctr3, ctr4, ctr5; }
    locations {
        s0 <ini: T>,
        s1 <ini: F>;
    }
    transitions {
        (s0, a, [(x, ==, 5)], [x], [ctr5 = 1 - ctr5], s1),
        (s1, a, [(x, ==, 5)], [x], [ctr5 = 1 - ctr5], s0);
    }
}
