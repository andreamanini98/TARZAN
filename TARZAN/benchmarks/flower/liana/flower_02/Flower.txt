create automaton Flower
{
    clocks { x1, x2, y; }
    actions { a; }
    locations {
        Start <ini: T>,
        Goal  <ini: F>;
    }
    transitions {
        (Start, a, [(x2, ==, 2)], [x2], Start),
        (Start, a, [(x1, ==, 1)], [x1], Start),
        (Start, a, [(x1, ==, 0), (x2, ==, 0), (y, >=, 1)], [], Goal);
    }
}
