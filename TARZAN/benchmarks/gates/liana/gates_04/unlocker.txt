create automaton unlocker
{
    clocks { x; }
    actions { a; }
    integers { gate; }
    locations {
        G0   <ini: T>,
        G1   <ini: F>,
        G2   <ini: F>,
        G3   <ini: F>,
        G4   <ini: F>,
        goal <ini: F>;
    }
    transitions {
        (G0, a, [(x, ==, 4)], [x], G1),
        (G1, a, [(x, ==, 4)], [x], G2),
        (G2, a, [(x, ==, 4)], [x], G3),
        (G3, a, [(x, ==, 4)], [x], G4),
        (G4, a, [(x, ==, 4)], gate == 4, [x], goal);
    }
}
