create automaton P42
{
    clocks { x; }
    actions { begin, busy, cd42, end; }
    locations {
        sender_wait   <ini: T                     >,
        sender_transm <ini: F, inv: [(x, <=, 808)]>,
        sender_retry  <ini: F, inv: [(x, <, 52)]  >;
    }
    transitions {
        (sender_wait, begin!, [], [x], sender_transm),
        (sender_wait, cd42?, [], [x], sender_wait),
        (sender_wait, cd42?, [], [x], sender_retry),
        (sender_wait, busy?, [], [x], sender_retry),
        (sender_transm, end!, [(x, ==, 808)], [x], sender_wait),
        (sender_transm, cd42?, [(x, <, 52)], [x], sender_retry),
        (sender_retry, begin!, [(x, <, 52)], [x], sender_transm),
        (sender_retry, cd42?, [(x, <, 52)], [x], sender_retry);
    }
}
